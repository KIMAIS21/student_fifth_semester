def fib(n):
    a, b = 1, 1
    for _ in range(n):
        yield a
        a, b = b, a + b

fibonacci_numbers = list(fib(200))
print(fibonacci_numbers)

def fib(n):
    a, b = 1, 1
    with open('fib.txt', 'w') as file:
        for _ in range(n):
            file.write(str(a) + '\n')
            a, b = b, a + b
            yield a
    with open('fib.txt', 'r') as file:
        print(file.readlines()[-1])
fibonacci_numbers = list(fib(200))
